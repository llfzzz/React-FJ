/* Free Joy — ScrollProgress.css
   Pairs with ScrollProgress.tsx / ScrollProgress.jsx. Requires the FJ tokens
   (tokens/*.css).
   The component drives the fill by setting scaleX from the live scroll
   position and feeds --fj-height, --fj-color, and --fj-z. Pinned to the top
   by default; --bottom pins low, --sticky tracks a scrollable container.
   (Prefer a scroll listener over animation-timeline while browser support
   is partial — the JS is ~10 lines and works everywhere.) */

.fj-scrollbar {
  position: fixed;
  top: 0;
  left: 0;
  right: 0;
  height: var(--fj-height, 3px);
  z-index: var(--fj-z, var(--z-sticky));
  background: var(--fj-color, var(--accent));
  transform-origin: left;
  transform: scaleX(0);
  pointer-events: none;
}

.fj-scrollbar--bottom { top: auto; bottom: 0; }

/* Tracking a scrollable container instead of the window. */
.fj-scrollbar--sticky { position: sticky; }
