// Free Joy — Highlighter (effect) with Tailwind CSS.
// Utilities reference the FJ tokens (arbitrary values) so both themes work.
// The scroll trigger is JS (IntersectionObserver toggling `lit`); the swipe is
// a background-size transition.
import { useState } from "react";

const TONES = { sun: "var(--sun-200)", coral: "var(--joy-100)", bloom: "var(--bloom-200)" };

/**
 * Free Joy — Highlighter (effect)
 * A marker swipe fills in behind a phrase. (Wire `lit` to an IntersectionObserver
 * for the scroll trigger.)
 */
export function Highlighter({ children, color = "sun", duration = 480, className = "" }) {
  const [lit, setLit] = useState(false);
  const tone = TONES[color] || color;
  return (
    <mark
      onMouseEnter={() => setLit(true)}
      style={{
        backgroundImage: `linear-gradient(${tone}, ${tone})`,
        backgroundSize: lit ? "100% 100%" : "0% 100%",
        transitionDuration: `${duration}ms`,
      }}
      className={
        "inline rounded-[var(--radius-xs)] bg-transparent bg-left bg-no-repeat px-[0.15em] " +
        "-mx-[0.15em] text-inherit transition-[background-size] ease-[var(--ease-out)] " +
        "motion-reduce:transition-none " +
        className
      }
    >
      {children}
    </mark>
  );
}
