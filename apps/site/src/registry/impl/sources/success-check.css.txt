/* Free Joy — SuccessCheck.css
   Pairs with SuccessCheck.tsx / SuccessCheck.jsx. Requires the FJ tokens
   (tokens/*.css).
   The checkmark path draws itself in via stroke-dashoffset, and a soft ring
   pulses out once, when the component adds .is-drawn. The component feeds
   --fj-size, --fj-color, and --fj-duration. */

.fj-check {
  position: relative;
  display: inline-flex;
  width: var(--fj-size, 48px);
  height: var(--fj-size, 48px);
  color: var(--fj-color, var(--success-500));
}

.fj-check svg { width: 100%; height: 100%; }

.fj-check-circle {
  fill: none;
  stroke: currentColor;
  stroke-opacity: 0.2;
  stroke-width: 2;
}

.fj-check-mark {
  fill: none;
  stroke: currentColor;
  stroke-width: 3.5;
  stroke-linecap: round;
  stroke-linejoin: round;
  stroke-dasharray: 28;
  stroke-dashoffset: 28;
  transition: stroke-dashoffset var(--fj-duration, 500ms) var(--ease-out);
}

.fj-check.is-drawn .fj-check-mark { stroke-dashoffset: 0; }

.fj-check-ring {
  position: absolute;
  inset: 0;
  border-radius: 50%;
  border: 2px solid currentColor;
  opacity: 0;
}
.fj-check.is-drawn .fj-check-ring {
  animation: fj-check-ring 700ms var(--ease-out) forwards;
}

@keyframes fj-check-ring {
  from { transform: scale(0.7); opacity: 0.6; }
  to   { transform: scale(1.15); opacity: 0; }
}

@media (prefers-reduced-motion: reduce) {
  .fj-check-mark { transition: none; stroke-dashoffset: 0; }
  .fj-check-ring { display: none; }
}
