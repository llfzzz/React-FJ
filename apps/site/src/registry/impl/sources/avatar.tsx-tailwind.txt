// Free Joy — Avatar (TypeScript + Tailwind CSS)
// Utilities reference the FJ tokens (arbitrary values), so every style stays token-driven.
import * as React from "react";

export interface AvatarProps extends React.HTMLAttributes<HTMLSpanElement> {
  /** Image URL. Falls back to initials when absent. */
  src?: string;
  /** Full name — used for initials and alt text. */
  name?: string;
  /** @default "md" */
  size?: "sm" | "md" | "lg";
}

const SIZES: Record<NonNullable<AvatarProps["size"]>, string> = {
  sm: "size-7 text-[10px]",
  md: "size-10 text-sm",
  lg: "size-14 text-xl",
};

/**
 * Free Joy — Avatar
 * Image or initials, circular. sizes: sm | md | lg
 */
export function Avatar({ src, name = "", size = "md", className = "", ...rest }: AvatarProps) {
  const initials = name
    .split(" ")
    .map((w) => w[0])
    .slice(0, 2)
    .join("")
    .toUpperCase();
  return (
    <span
      className={
        "inline-flex flex-none items-center justify-center overflow-hidden rounded-full " +
        "bg-[var(--joy-100)] text-[var(--joy-700)] " +
        "font-[family-name:var(--font-text)] font-semibold " +
        `${SIZES[size] ?? SIZES.md} ${className}`
      }
      {...rest}
    >
      {src ? <img src={src} alt={name} className="size-full object-cover" /> : initials || "?"}
    </span>
  );
}
