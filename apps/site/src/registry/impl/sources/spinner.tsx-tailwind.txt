// Free Joy — Spinner (TypeScript + Tailwind CSS)
// Utilities reference the FJ tokens (arbitrary values), so every style stays token-driven.
// Uses Tailwind's built-in animate-spin; motion-reduce slows it rather than
// freezing so it still reads as "busy".
import * as React from "react";

export interface SpinnerProps extends React.HTMLAttributes<HTMLSpanElement> {
  /** Pixel diameter. @default 22 */
  size?: number;
  /** @default "var(--accent)" */
  color?: string;
  /** Ring stroke width. @default 2.5 */
  thickness?: number;
  /** Optional text shown beside the ring. */
  label?: string;
}

/**
 * Free Joy — Spinner
 * Minimal loading indicator. Inherits color; size in px.
 */
export function Spinner({
  size = 22,
  color = "var(--accent)",
  thickness = 2.5,
  label,
  className = "",
  style,
  ...rest
}: SpinnerProps) {
  return (
    <span
      role="status"
      aria-label={label || "Loading"}
      style={{ color, ...style }}
      className={`inline-flex items-center gap-[9px] ${className}`}
      {...rest}
    >
      <span
        style={{ width: size, height: size, borderWidth: thickness }}
        className={
          "flex-none rounded-full border-current border-t-transparent opacity-90 " +
          "animate-spin [animation-duration:0.7s] motion-reduce:[animation-duration:2.4s]"
        }
      />
      {label && <span className="text-[length:var(--text-sm)] text-[var(--text-muted)]">{label}</span>}
    </span>
  );
}
