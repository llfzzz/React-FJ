/* Free Joy — AnimatedBorder.css
   Pairs with AnimatedBorder.tsx / AnimatedBorder.jsx. Requires the FJ tokens
   (tokens/*.css).
   Fully CSS: an animated @property angle drives a conic-gradient border that
   sweeps around the panel. The component feeds --fj-speed, --fj-width, and
   --fj-aborder-colors. Reserve for ONE flagship element per view. */

/* Registering the angle as an animatable custom property is what lets the
   conic gradient rotate smoothly. Unsupported browsers show it static. */
@property --fj-angle {
  syntax: "<angle>";
  initial-value: 0deg;
  inherits: false;
}
@keyframes fj-angle-spin { to { --fj-angle: 360deg; } }

.fj-aborder {
  position: relative;
  padding: var(--fj-width, 1px);
  border-radius: var(--radius-lg);
  background: conic-gradient(
    from var(--fj-angle, 0deg),
    var(--border) 0%,
    var(--fj-aborder-colors, var(--joy-400), var(--bloom-300), var(--sun-300)),
    var(--border) 100%
  );
  animation: fj-angle-spin var(--fj-speed, 8s) linear infinite;
}

.fj-aborder-inner {
  position: relative;
  background: var(--surface);
  /* Inner radius = outer radius minus the border width. */
  border-radius: calc(var(--radius-lg) - var(--fj-width, 1px));
  padding: var(--space-5);
}

/* Reduced motion: hold the gradient still (still a pretty static border). */
@media (prefers-reduced-motion: reduce) {
  .fj-aborder { animation: none; }
}
