/* Free Joy — StaggerList.css
   Pairs with StaggerList.tsx / StaggerList.jsx. Requires the FJ tokens
   (tokens/*.css).
   Children rise + fade in with a per-item delay set from --fj-i; the component
   adds .is-in on scroll into view and feeds --fj-stagger, --fj-duration, and
   --fj-delay. Entry styles: up (default) | --fade | --scale. Container layout
   is yours — the wrapper imposes none. */

.fj-stagger-item {
  opacity: 0;
  transform: translateY(14px);
  transition:
    opacity var(--fj-duration, 480ms) var(--ease-emphasized),
    transform var(--fj-duration, 480ms) var(--ease-emphasized);
  transition-delay: calc(var(--fj-delay, 0ms) + var(--fj-i, 0) * var(--fj-stagger, 60ms));
}

.fj-stagger--fade .fj-stagger-item { transform: none; }
.fj-stagger--scale .fj-stagger-item { transform: scale(0.96); }

.fj-stagger.is-in .fj-stagger-item {
  opacity: 1;
  transform: none;
}

@media (prefers-reduced-motion: reduce) {
  .fj-stagger-item { opacity: 1; transform: none; transition: none; }
}
