// global stylesheet once (Tailwind's built-in `animate-ping` is close, but
// this echo is softer and slower, tuned to FJ motion).

/**
 * Free Joy — PingDot (effect)
 * A notification dot with a radiating echo, pinned to an anchor's corner.
 */
export function PingDot({ children, size = 10, color = "var(--accent)", label = "New activity", className = "" }) {
  const dot = (
    <span
      role="status"
      aria-label={label}
      style={{ width: size, height: size, background: color }}
      className={children ? "absolute right-0 top-0 inline-block translate-x-[35%] -translate-y-[35%] rounded-full" : "relative inline-block rounded-full align-middle"}
    >
      <span
        aria-hidden="true"
        style={{ background: color }}
        className="absolute inset-0 rounded-full animate-[fj-ping_1800ms_var(--ease-out)_infinite] motion-reduce:hidden"
      />
    </span>
  );
  if (!children) return dot;
  return (
    <span className={`relative inline-block ${className}`}>
      {children}
      {dot}
    </span>
  );
}
