// Free Joy — Waves (effect) (TypeScript + Tailwind CSS)
// The slide keyframes can't be utilities — add the CSS block below to your
// global stylesheet once. The path spans two 1440-unit periods and the
// keyframe slides exactly one, so the loop is seamless. Only transform animates.
"use client";

import * as React from "react";

const PERIOD = 1440;
const VIEW_H = 120;
const MAX_LAYERS = 4;
const MAX_AMPLITUDE = 24;
const OPACITIES = [0.38, 0.3, 0.22, 0.14];

/** A smooth two-period sine-ish stroke path (Q + reflected T segments). */
function wavePath(amplitude: number): string {
  const y = VIEW_H / 2;
  const half = PERIOD / 4;
  let d = `M 0 ${y} Q ${half / 2} ${y - amplitude} ${half} ${y}`;
  for (let x = half * 2; x <= PERIOD * 2; x += half) {
    d += ` T ${x} ${y}`;
  }
  return d;
}

export interface WavesProps extends React.HTMLAttributes<HTMLDivElement> {
  children?: React.ReactNode;
  /** Wave-line count (capped at 4; "lite" caps at 2). @default 3 */
  layers?: number;
  /** Line color. @default var(--joy-300) */
  color?: string;
  /** Wave height in viewBox units (capped at 24). @default 14 */
  amplitude?: number;
  /** Band height in px, anchored to the container bottom. @default 120 */
  height?: number;
  /** Speed multiplier (1 = default). @default 1 */
  speed?: number;
  /** "lite" caps the layer count at 2. @default "full" */
  performance?: "full" | "lite";
  /** Static lines, no drift. @default false */
  disabled?: boolean;
}

/**
 * Free Joy — Waves (effect)
 * Slow wave lines drifting across a band at the container's bottom edge — a
 * quiet texture for heroes and footers.
 */
export function Waves({
  children,
  layers = 3,
  color = "var(--joy-300)",
  amplitude = 14,
  height = 120,
  speed = 1,
  performance = "full",
  disabled = false,
  className = "",
  style,
  ...rest
}: WavesProps) {
  const n = Math.min(performance === "lite" ? 2 : MAX_LAYERS, Math.max(1, layers));
  const d = React.useMemo(() => wavePath(Math.min(amplitude, MAX_AMPLITUDE)), [amplitude]);

  return (
    <div className={`relative overflow-hidden ${className}`} style={style} {...rest}>
      <svg
        aria-hidden="true"
        viewBox={`0 0 ${PERIOD} ${VIEW_H}`}
        preserveAspectRatio="none"
        style={{ height }}
        className="pointer-events-none absolute inset-x-0 bottom-0 block w-full"
      >
        {Array.from({ length: n }, (_, i) => (
          <g
            key={i}
            style={{
              animationDuration: `${(12 + i * 5) / speed}s`,
              animationDirection: i % 2 === 1 ? "reverse" : undefined,
            }}
            className={
              "will-change-transform " +
              (disabled ? "" : "animate-[fj-wave_linear_infinite] motion-reduce:animate-none")
            }
          >
            <path
              d={d}
              transform={`translate(0 ${(i - (n - 1) / 2) * 14})`}
              fill="none"
              stroke={color}
              strokeWidth={2}
              strokeOpacity={OPACITIES[i]}
              vectorEffect="non-scaling-stroke"
            />
          </g>
        ))}
      </svg>
      <div className="relative">{children}</div>
    </div>
  );
}

/* Add once to your global CSS (slides one 1440-unit period):
@keyframes fj-wave {
  from { transform: translateX(0); }
  to { transform: translateX(-1440px); }
}
