/* Free Joy — Shimmer.css
   Pairs with Shimmer.tsx / Shimmer.jsx. Requires the FJ tokens (tokens/*.css).
   A sheen overlay sweeps across the wrapped content. The wrapper clips it and
   stays the size of its child. The component feeds --fj-duration, --fj-angle,
   and --fj-intensity. Modifiers: --hover (sweep on hover only), --once (one
   sweep on mount). */

.fj-shimmer {
  position: relative;
  display: inline-block;
  overflow: hidden;
}

.fj-shimmer-sheen {
  position: absolute;
  inset: 0;
  pointer-events: none;
  transform: translateX(-100%);
  background: linear-gradient(
    var(--fj-angle, 110deg),
    transparent 20%,
    color-mix(in srgb, var(--white) calc(var(--fj-intensity, 0.35) * 100%), transparent) 50%,
    transparent 80%
  );
  animation: fj-shimmer-sweep var(--fj-duration, 1400ms) var(--ease-in-out) infinite;
}

/* Hover-only variant: pause until hovered. */
.fj-shimmer--hover .fj-shimmer-sheen { animation-play-state: paused; }
.fj-shimmer--hover:hover .fj-shimmer-sheen { animation-play-state: running; }

/* Mount variant: a single sweep. */
.fj-shimmer--once .fj-shimmer-sheen { animation-iteration-count: 1; }

@keyframes fj-shimmer-sweep {
  from { transform: translateX(-100%); }
  to   { transform: translateX(100%); }
}

@media (prefers-reduced-motion: reduce) {
  .fj-shimmer-sheen { animation: none; opacity: 0; }
}
