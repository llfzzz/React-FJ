// Free Joy — Input (TypeScript + Tailwind CSS)
// Utilities reference the FJ tokens (arbitrary values), so every style stays token-driven.
import * as React from "react";

export interface InputProps extends Omit<React.InputHTMLAttributes<HTMLInputElement>, "size"> {
  /** Field label rendered above the input. */
  label?: string;
  /** Helper text below the field. */
  hint?: string;
  /** Error message — turns the field red and overrides hint. */
  error?: string;
  /** Icon node rendered inside, before the text. */
  iconLeft?: React.ReactNode;
  /** @default "md" */
  size?: "sm" | "md" | "lg";
}

const SIZES: Record<NonNullable<InputProps["size"]>, string> = {
  sm: "h-9",
  md: "h-11",
  lg: "h-13",
};

/**
 * Free Joy — Input
 * Text field with optional label, hint, error, and leading icon.
 * Focus styling uses focus-within so a leading icon sits inside the glow.
 */
export function Input({ label, hint, error, iconLeft, size = "md", id, className = "", ...rest }: InputProps) {
  const autoId = React.useId();
  const fid = id || autoId;
  return (
    <div className={`flex flex-col gap-1.5 ${className}`}>
      {label && (
        <label
          htmlFor={fid}
          className="text-[length:var(--text-sm)] font-medium text-[var(--text-muted)]"
        >
          {label}
        </label>
      )}
      <div
        className={
          "flex items-center gap-2 rounded-[var(--radius-md)] bg-[var(--surface)] px-3.5 " +
          "border transition-[border-color,box-shadow] duration-[var(--dur-fast)] ease-[var(--ease-out)] " +
          `${SIZES[size] ?? SIZES.md} ` +
          (error
            ? "border-[var(--danger-500)]"
            : "border-[var(--border-strong)] focus-within:border-[var(--accent)] focus-within:shadow-[var(--ring)]")
        }
      >
        {iconLeft && <span className="inline-flex text-[var(--text-subtle)]">{iconLeft}</span>}
        <input
          id={fid}
          className={
            "min-w-0 flex-1 border-none bg-transparent outline-none " +
            "font-[family-name:var(--font-text)] text-[length:var(--text-base)] text-[var(--text)]"
          }
          {...rest}
        />
      </div>
      {(hint || error) && (
        <span
          className={`text-[length:var(--text-xs)] ${
            error ? "text-[var(--danger-700)]" : "text-[var(--text-subtle)]"
          }`}
        >
          {error || hint}
        </span>
      )}
    </div>
  );
}
