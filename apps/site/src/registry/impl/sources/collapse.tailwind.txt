// Free Joy — Collapse (effect) with Tailwind CSS.
// Uses the modern grid-template-rows 0fr → 1fr trick — pure CSS height
// animation, no JS measuring. `open` toggles the row from 0fr to 1fr.

/**
 * Free Joy — Collapse (effect)
 * Smoothly expands/collapses to the content's natural height.
 */
export function Collapse({ children, open, duration = 280, className = "" }) {
  return (
    <div
      aria-hidden={!open}
      style={{ transitionDuration: `${duration}ms` }}
      className={
        "grid transition-[grid-template-rows] ease-[var(--ease-out)] motion-reduce:transition-none " +
        (open ? "grid-rows-[1fr]" : "grid-rows-[0fr]") +
        " " +
        className
      }
    >
      {/* min-h-0 lets the row collapse fully; overflow-hidden clips during travel */}
      <div className="min-h-0 overflow-hidden">{children}</div>
    </div>
  );
}
