/* Free Joy — Orbs.css
   Pairs with Orbs.tsx / Orbs.jsx. Requires the FJ tokens (tokens/*.css).
   Soft gradient blobs drifting slowly behind the content. The component
   scatters each orb's position, size, color, and per-orb duration/delay
   inline; this stylesheet carries the shared blur, opacity, and drift loop
   (--fj-blur / --fj-opacity from props). Only transform animates; --static
   holds the wash still. */

.fj-orbs {
  --fj-blur: 40px;
  --fj-opacity: 0.5;
  position: relative;
  overflow: hidden;
}

.fj-orb {
  position: absolute;
  border-radius: 50%;
  filter: blur(var(--fj-blur));
  opacity: var(--fj-opacity);
  pointer-events: none;
  will-change: transform;
  animation-name: fj-orb;
  animation-timing-function: var(--ease-in-out);
  animation-iteration-count: infinite;
  animation-duration: 24s; /* per-orb duration/delay set inline override this */
}

.fj-orbs--static .fj-orb { animation: none; }

.fj-orbs-content { position: relative; }

@keyframes fj-orb {
  0% { transform: translate3d(0, 0, 0); }
  33% { transform: translate3d(6%, -8%, 0); }
  66% { transform: translate3d(-5%, 6%, 0); }
  100% { transform: translate3d(0, 0, 0); }
}

/* Reduced motion: the wash stays, the drift stops. */
@media (prefers-reduced-motion: reduce) {
  .fj-orb { animation: none; }
}
