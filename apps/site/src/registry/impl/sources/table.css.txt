/* Free Joy — Table.css
   Pairs with Table.tsx / Table.jsx. Requires the FJ tokens (tokens/*.css).
   Modifiers: .fj-table--zebra (alternating tints), .fj-table--dense (tighter
   rows), .fj-table--no-hover (static rows). Row hover is pure CSS. */

.fj-table-wrap {
  width: 100%;
  overflow: auto;
  border: 1px solid var(--border);
  border-radius: var(--radius-lg);
  background: var(--surface);
}

.fj-table {
  width: 100%;
  border-collapse: collapse;
  font-family: var(--font-text);
}

.fj-table th {
  text-align: left;
  padding: 14px 18px;
  font-size: var(--text-xs);
  font-weight: var(--weight-semibold);
  text-transform: uppercase;
  letter-spacing: 0.04em;
  color: var(--text-subtle);
  white-space: nowrap;
  background: var(--surface-2);
  border-bottom: 1px solid var(--border);
}

.fj-table td {
  padding: 14px 18px;
  font-size: var(--text-sm);
  color: var(--text);
  border-bottom: 1px solid var(--border);
}
.fj-table tbody tr:last-child td { border-bottom: none; }

/* Zebra + dense modifiers (declared before hover so hover wins on even rows) */
.fj-table--zebra tbody tr:nth-child(even) { background: var(--surface-2); }
.fj-table--dense th, .fj-table--dense td { padding: 10px 14px; }

/* Row hover */
.fj-table tbody tr {
  transition: background var(--dur-fast) var(--ease-out);
}
.fj-table tbody tr:hover { background: var(--surface-hover); }
.fj-table--no-hover tbody tr:hover {
  background: transparent;
}
.fj-table--no-hover.fj-table--zebra tbody tr:nth-child(even):hover {
  background: var(--surface-2);
}

@media (prefers-reduced-motion: reduce) {
  .fj-table tbody tr { transition-duration: 0.01ms; }
}
