// Free Joy — Tactile (effect) (TypeScript + Tailwind CSS)
// Pure :active press — no JS needed. Wrap any child. Reduced motion keeps the
// press instant with no scaling.
import * as React from "react";

export interface TactileProps extends React.HTMLAttributes<HTMLSpanElement> {
  children?: React.ReactNode;
  /** Scale while pressed. @default 0.96 */
  scale?: number;
  /** Press transition duration in ms. @default 120 */
  duration?: number;
  /** Disable (children unchanged). @default false */
  disabled?: boolean;
}

/**
 * Free Joy — Tactile (effect)
 * Adds a satisfying press-down (scale + settle) to any child — buttons, cards,
 * tappable tiles.
 */
export function Tactile({
  children,
  scale = 0.96,
  duration = 120,
  disabled = false,
  className = "",
  style,
  ...rest
}: TactileProps) {
  return (
    <span
      style={{ "--fj-scale": scale, transitionDuration: `${duration}ms`, ...style } as React.CSSProperties}
      className={
        "inline-block [-webkit-tap-highlight-color:transparent] " +
        "transition-transform ease-[var(--ease-out)] " +
        (disabled
          ? "transition-none "
          : "active:scale-[var(--fj-scale)] motion-reduce:transition-none motion-reduce:active:scale-100 ") +
        className
      }
      {...rest}
    >
      {children}
    </span>
  );
}
