/* Free Joy — Float.css
   Pairs with Float.tsx / Float.jsx. Requires the FJ tokens (tokens/*.css).
   A gentle levitation loop. The component feeds --fj-distance, --fj-duration,
   --fj-delay, and --fj-rotate; --off holds it static. Static under reduced
   motion. */

.fj-float {
  display: inline-block;
  animation: fj-float var(--fj-duration, 4s) var(--ease-in-out) var(--fj-delay, 0s) infinite;
  will-change: transform;
}

.fj-float--off { animation: none; will-change: auto; }

@keyframes fj-float {
  0%, 100% { transform: translateY(0) rotate(0deg); }
  50%      { transform: translateY(calc(-1 * var(--fj-distance, 8px))) rotate(var(--fj-rotate, 0deg)); }
}

@media (prefers-reduced-motion: reduce) {
  .fj-float { animation: none; }
}
