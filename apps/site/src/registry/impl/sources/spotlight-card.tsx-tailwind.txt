// Free Joy — SpotlightCard (effect) (TypeScript + Tailwind CSS)
// Utilities reference the FJ tokens (arbitrary values), so every style stays token-driven.
// The glow follows the cursor via --fj-mx / --fj-my set in a mousemove handler;
// the radial paint itself is an arbitrary background value.
"use client";

import * as React from "react";

export interface SpotlightCardProps extends React.HTMLAttributes<HTMLDivElement> {
  /** Spotlight diameter in px. @default 320 */
  size?: number;
  /** @default "var(--space-5)" */
  padding?: number | string;
  /** @default "var(--radius-lg)" */
  radius?: number | string;
  /** Lift + spotlight on hover. @default true */
  interactive?: boolean;
}

/**
 * Free Joy — SpotlightCard (effect)
 * A Card whose surface carries a soft warm highlight that follows the cursor.
 * For flagship/showcase cards only — not dense data views.
 */
export function SpotlightCard({
  size = 320,
  padding,
  radius,
  interactive = true,
  className = "",
  style,
  children,
  ...rest
}: SpotlightCardProps) {
  const ref = React.useRef<HTMLDivElement>(null);
  const onMove = (e: React.MouseEvent<HTMLDivElement>) => {
    const el = ref.current;
    if (!el) return;
    const r = el.getBoundingClientRect();
    el.style.setProperty("--fj-mx", `${e.clientX - r.left}px`);
    el.style.setProperty("--fj-my", `${e.clientY - r.top}px`);
  };
  return (
    <div
      ref={ref}
      onMouseMove={onMove}
      style={{ padding, borderRadius: radius, ...style }}
      className={
        "group relative overflow-hidden rounded-[var(--radius-lg)] border border-[var(--border)] " +
        "bg-[var(--surface)] p-[var(--space-5)] shadow-[var(--shadow-xs)] " +
        "transition-[box-shadow,transform] duration-[var(--dur-base)] ease-[var(--ease-out)] " +
        (interactive ? "hover:-translate-y-0.5 hover:shadow-[var(--shadow-md)] motion-reduce:hover:translate-y-0 " : "") +
        className
      }
      {...rest}
    >
      {interactive && (
        <div
          aria-hidden="true"
          style={{
            background: `radial-gradient(${size}px circle at var(--fj-mx,50%) var(--fj-my,40%), var(--accent-soft), transparent 70%)`,
          }}
          className="pointer-events-none absolute inset-0 opacity-0 transition-opacity duration-[var(--dur-base)] ease-[var(--ease-out)] group-hover:opacity-100"
        />
      )}
      <div className="relative">{children}</div>
    </div>
  );
}
