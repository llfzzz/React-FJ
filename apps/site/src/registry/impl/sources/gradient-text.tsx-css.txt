// Free Joy — GradientText (effect) (TypeScript + CSS)
// Pairs with GradientText.css (imported below). Requires the FJ tokens (tokens/*.css).
// The gradient is clipped to the text and slid on a loop; reduced motion holds
// it still (the color still reads).
import * as React from "react";
import "./GradientText.css";

export interface GradientTextProps extends React.HTMLAttributes<HTMLSpanElement> {
  /** The text (or nodes) to paint with the moving gradient. */
  children?: React.ReactNode;
  /** Gradient color stops (FJ tokens or any CSS colors). @default coral→bloom→sun */
  colors?: string[];
  /** Seconds per loop. @default 6 */
  duration?: number;
  /** Gradient angle in degrees. @default 100 */
  angle?: number;
  /** Keep the gradient animating. @default true */
  loop?: boolean;
  /** Render plain text with no gradient. @default false */
  disabled?: boolean;
}

/**
 * Free Joy — GradientText (effect)
 * A word or heading painted with a slow-moving FJ gradient. Landing/hero
 * surfaces. Under reduced motion (or disabled) the gradient holds still.
 */
export function GradientText({
  children,
  colors = ["var(--joy-500)", "var(--bloom-500)", "var(--sun-500)"],
  duration = 6,
  angle = 100,
  loop = true,
  disabled = false,
  className = "",
  style,
  ...rest
}: GradientTextProps) {
  if (disabled) {
    return (
      <span className={className} style={style} {...rest}>
        {children}
      </span>
    );
  }
  const stops = [...colors, colors[0]].join(", ");
  const vars = {
    "--fj-gradient-stops": stops,
    "--fj-angle": `${angle}deg`,
    "--fj-duration": `${duration}s`,
  } as React.CSSProperties;

  const classes = ["fj-gradient-text", !loop && "fj-gradient-text--still", className]
    .filter(Boolean)
    .join(" ");

  return (
    <span className={classes} style={{ ...vars, ...style }} {...rest}>
      {children}
    </span>
  );
}
