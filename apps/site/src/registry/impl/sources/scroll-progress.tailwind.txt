// Free Joy — ScrollProgress (effect) with Tailwind CSS.
// The fill needs the scroll position, which is JavaScript state (or the modern
// scroll-driven-animation CSS). Utilities handle the layout + the scaleX; the
// value comes from a small scroll listener.
import { useEffect, useState } from "react";

/**
 * Free Joy — ScrollProgress (effect)
 * A thin reading-progress bar pinned to the viewport edge that fills on scroll.
 */
export function ScrollProgress({ height = 3, color = "var(--accent)", position = "top" }) {
  const [progress, setProgress] = useState(0);
  useEffect(() => {
    const update = () => {
      const max = document.documentElement.scrollHeight - window.innerHeight;
      setProgress(max > 0 ? window.scrollY / max : 0);
    };
    update();
    window.addEventListener("scroll", update, { passive: true });
    window.addEventListener("resize", update);
    return () => {
      window.removeEventListener("scroll", update);
      window.removeEventListener("resize", update);
    };
  }, []);
  return (
    <div
      role="progressbar"
      aria-valuenow={Math.round(progress * 100)}
      aria-valuemin={0}
      aria-valuemax={100}
      aria-label="Reading progress"
      style={{ height, background: color, transform: `scaleX(${progress})` }}
      className={`pointer-events-none fixed inset-x-0 z-[var(--z-sticky)] origin-left ${position === "bottom" ? "bottom-0" : "top-0"}`}
    />
  );
}
