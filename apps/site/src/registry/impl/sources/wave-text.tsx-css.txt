// Free Joy — WaveText (effect) (TypeScript + CSS)
// Pairs with WaveText.css (imported below). Requires the FJ tokens (tokens/*.css).
// The string splits into aria-hidden per-character spans that take turns
// rising; the root keeps the full text as one aria-label. Reduced motion
// holds the wave still (see the CSS media query).
import * as React from "react";
import "./WaveText.css";

export interface WaveTextProps extends Omit<React.HTMLAttributes<HTMLSpanElement>, "children"> {
  /** The text to animate, split per character. */
  text: string;
  /** When the wave plays. @default "loop" */
  trigger?: "loop" | "mount" | "inview";
  /** Rise height in px (kept small). @default 6 */
  distance?: number;
  /** ms per character cycle. @default 1200 */
  duration?: number;
  /** ms stagger between adjacent characters. @default 60 */
  stagger?: number;
  /** Plain text, no wave. @default false */
  disabled?: boolean;
}

/**
 * Free Joy — WaveText (effect)
 * Characters take turns rising in a gentle wave — playful hero words and
 * section accents.
 */
export function WaveText({
  text,
  trigger = "loop",
  distance = 6,
  duration = 1200,
  stagger = 60,
  disabled = false,
  className = "",
  style,
  ...rest
}: WaveTextProps) {
  const ref = React.useRef<HTMLSpanElement>(null);
  const [inView, setInView] = React.useState(trigger !== "inview");

  React.useEffect(() => {
    if (trigger !== "inview") return;
    const el = ref.current;
    if (!el || typeof IntersectionObserver === "undefined") {
      setInView(true);
      return;
    }
    const io = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            setInView(true);
            io.disconnect();
          }
        });
      },
      { threshold: 0.2 },
    );
    io.observe(el);
    return () => io.disconnect();
  }, [trigger]);

  if (disabled) {
    return (
      <span className={className} style={style} {...rest}>
        {text}
      </span>
    );
  }

  const classes = [
    "fj-wave-text",
    trigger !== "loop" && "fj-wave-text--once",
    !inView && "fj-wave-text--waiting",
    className,
  ]
    .filter(Boolean)
    .join(" ");

  const vars = {
    "--fj-wave-distance": `${distance}px`,
    "--fj-duration": `${duration}ms`,
    "--fj-stagger": `${stagger}ms`,
  } as React.CSSProperties;

  return (
    <span ref={ref} role="text" aria-label={text} className={classes} style={{ ...vars, ...style }} {...rest}>
      <span aria-hidden="true">
        {Array.from(text).map((ch, i) => (
          <span key={i} className="fj-wave-text__char" style={{ "--fj-i": i } as React.CSSProperties}>
            {ch}
          </span>
        ))}
      </span>
    </span>
  );
}
