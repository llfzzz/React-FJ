/* Free Joy — TextReveal.css
   Pairs with TextReveal.tsx / TextReveal.jsx. Requires the FJ tokens (tokens/*.css).
   The component adds .is-in when the text scrolls into view; each word carries
   its own --fj-i (index) for the stagger delay. --fj-delay / --fj-stagger /
   --fj-duration arrive from props. */

.fj-reveal {
  --fj-delay: 0ms;
  --fj-stagger: 45ms;
  --fj-duration: 640ms;
  display: inline-block;
  white-space: pre-wrap;
}

/* Clip box: the inner span rises up from below the baseline. */
.fj-reveal-w {
  display: inline-block;
  overflow: hidden;
  vertical-align: bottom;
}
.fj-reveal-w > span {
  display: inline-block;
  white-space: pre;
  transform: translateY(105%);
  opacity: 0;
  transition:
    transform var(--fj-duration) var(--ease-emphasized),
    opacity calc(var(--fj-duration) * 0.6) var(--ease-out);
  transition-delay: calc(var(--fj-delay) + var(--fj-i, 0) * var(--fj-stagger));
}

/* In view: everything rises to rest. */
.fj-reveal.is-in .fj-reveal-w > span {
  transform: translateY(0);
  opacity: 1;
}

/* Reduced motion: no rise, text simply present. */
@media (prefers-reduced-motion: reduce) {
  .fj-reveal-w > span { transform: none; opacity: 1; transition: none; }
}
