<!-- Free Joy — TextReveal (effect), no-framework version.
     Requires the FJ tokens (tokens/*.css).
     This reproduces the rise-in stagger on mount. To trigger it on SCROLL like
     the React component, add the .is-in class with an IntersectionObserver (a
     few lines of JS) — the animation itself is pure CSS below.
     Each word sets its own --fj-i (index) for the stagger delay. -->

<span class="fj-reveal" aria-label="Make something you love">
  <span class="fj-reveal-w" style="--fj-i: 0"><span>Make</span></span>
  <span class="fj-reveal-w" style="--fj-i: 1"><span>something</span></span>
  <span class="fj-reveal-w" style="--fj-i: 2"><span>you</span></span>
  <span class="fj-reveal-w" style="--fj-i: 3"><span>love</span></span>
</span>

<style>
.fj-reveal {
  --fj-stagger: 45ms;
  --fj-duration: 640ms;
  display: inline-block;
  white-space: pre-wrap;
}

/* Clip box: the inner span rises up from below the baseline. */
.fj-reveal-w {
  display: inline-block;
  overflow: hidden;
  vertical-align: bottom;
}
.fj-reveal-w > span {
  display: inline-block;
  white-space: pre;
  padding: 0 0.12em;
  transform: translateY(105%);
  opacity: 0;
  transition:
    transform var(--fj-duration) var(--ease-emphasized),
    opacity calc(var(--fj-duration) * 0.6) var(--ease-out);
  transition-delay: calc(var(--fj-i) * var(--fj-stagger));
}

/* Reveal on mount (swap for .is-in + IntersectionObserver to trigger on scroll). */
.fj-reveal > .fj-reveal-w > span {
  transform: translateY(0);
  opacity: 1;
}

/* Reduced motion: no rise, text simply present. */
@media (prefers-reduced-motion: reduce) {
  .fj-reveal-w > span { transform: none; opacity: 1; transition: none; }
}
</style>
