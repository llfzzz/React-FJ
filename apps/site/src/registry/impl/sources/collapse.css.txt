/* Free Joy — Collapse.css
   Pairs with Collapse.tsx / Collapse.jsx. Requires the FJ tokens (tokens/*.css).
   The modern grid-template-rows 0fr → 1fr trick animates to the content's
   natural height with pure CSS — no JS height measuring. The component toggles
   .is-open and feeds --fj-duration / --fj-ease. */

.fj-collapse {
  display: grid;
  grid-template-rows: 0fr;
  transition: grid-template-rows var(--fj-duration, 280ms) var(--fj-ease, var(--ease-out));
}

.fj-collapse.is-open {
  grid-template-rows: 1fr;
}

/* The inner wrapper must have min-height: 0 for the row to collapse fully. */
.fj-collapse-inner {
  overflow: hidden;
  min-height: 0;
}

@media (prefers-reduced-motion: reduce) {
  .fj-collapse { transition: none; }
}
