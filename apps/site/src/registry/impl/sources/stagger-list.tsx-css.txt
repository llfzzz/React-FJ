// Free Joy — StaggerList (effect) (TypeScript + CSS)
// Pairs with StaggerList.css (imported below). Requires the FJ tokens
// (tokens/*.css). Children rise + fade in with a per-item delay set from
// --fj-i; the scroll trigger toggles .is-in via IntersectionObserver. Under
// reduced motion (or disabled) all children appear at once.
"use client";

import * as React from "react";
import "./StaggerList.css";

export interface StaggerListProps extends React.HTMLAttributes<HTMLElement> {
  children?: React.ReactNode;
  /** Per-child delay in ms. @default 60 */
  stagger?: number;
  /** Per-child duration in ms. @default 480 */
  duration?: number;
  /** Entry style. @default "up" */
  from?: "up" | "fade" | "scale";
  /** Lead-in delay before the first child in ms. @default 0 */
  initialDelay?: number;
  /** Element/tag to render as the container. @default "div" */
  as?: React.ElementType;
  /** Disable (children appear immediately). @default false */
  disabled?: boolean;
}

/**
 * Free Joy — StaggerList (effect)
 * Reveals its direct children one after another when the list scrolls into
 * view. Great for feature grids and lists. Only transform/opacity animate.
 */
export function StaggerList({
  children,
  stagger = 60,
  duration = 480,
  from = "up",
  initialDelay = 0,
  as: Tag = "div",
  disabled = false,
  className = "",
  style,
  ...rest
}: StaggerListProps) {
  const ref = React.useRef<HTMLElement>(null);
  const [inView, setInView] = React.useState(false);

  React.useEffect(() => {
    const el = ref.current;
    if (!el) return;
    const io = new IntersectionObserver(
      ([e]) => {
        if (e.isIntersecting) {
          setInView(true);
          io.disconnect();
        }
      },
      { threshold: 0.2 },
    );
    io.observe(el);
    return () => io.disconnect();
  }, []);

  const classes = [
    "fj-stagger",
    from !== "up" && `fj-stagger--${from}`,
    (disabled || inView) && "is-in",
    className,
  ]
    .filter(Boolean)
    .join(" ");

  const vars = {
    "--fj-stagger": `${stagger}ms`,
    "--fj-duration": `${duration}ms`,
    "--fj-delay": `${initialDelay}ms`,
  } as React.CSSProperties;

  const items = React.Children.toArray(children);

  return (
    <Tag ref={ref} className={classes} style={{ ...vars, ...style }} {...rest}>
      {items.map((child, i) => (
        <div key={i} className="fj-stagger-item" style={{ "--fj-i": i } as React.CSSProperties}>
          {child}
        </div>
      ))}
    </Tag>
  );
}
