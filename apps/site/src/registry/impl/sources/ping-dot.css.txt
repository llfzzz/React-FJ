/* Free Joy — PingDot.css
   Pairs with PingDot.tsx / PingDot.jsx. Requires the FJ tokens (tokens/*.css).
   A solid dot with an aria-hidden echo that scales up and fades on a loop. The
   component feeds --fj-size, --fj-color, and --fj-duration, and picks a corner
   modifier (default top-right) or --inline. --static (or reduced motion) hides
   the echo; the solid dot keeps the meaning. */

.fj-ping-anchor {
  position: relative;
  display: inline-block;
}

.fj-ping {
  --fj-size: 10px;
  --fj-color: var(--accent);
  position: absolute;
  display: inline-block;
  width: var(--fj-size);
  height: var(--fj-size);
  border-radius: 50%;
  background: var(--fj-color);
}

/* Corner placements (with an anchor). */
.fj-ping--top-right    { top: 0; right: 0; transform: translate(35%, -35%); }
.fj-ping--top-left     { top: 0; left: 0; transform: translate(-35%, -35%); }
.fj-ping--bottom-right { bottom: 0; right: 0; transform: translate(35%, 35%); }
.fj-ping--bottom-left  { bottom: 0; left: 0; transform: translate(-35%, 35%); }

/* Inline (no anchor). */
.fj-ping--inline {
  position: relative;
  transform: none;
  vertical-align: middle;
}

.fj-ping-echo {
  position: absolute;
  inset: 0;
  border-radius: 50%;
  background: var(--fj-color);
  animation: fj-ping var(--fj-duration, 1800ms) var(--ease-out) infinite;
}

.fj-ping--static .fj-ping-echo { display: none; }

@keyframes fj-ping {
  from { transform: scale(1); opacity: 0.55; }
  to { transform: scale(2.6); opacity: 0; }
}

/* Reduced motion: no echo — the solid dot keeps the meaning. */
@media (prefers-reduced-motion: reduce) {
  .fj-ping-echo { display: none; }
}
