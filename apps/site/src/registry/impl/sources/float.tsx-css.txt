// Free Joy — Float (effect) (TypeScript + CSS)
// Pairs with Float.css (imported below). Requires the FJ tokens (tokens/*.css).
// Gentle levitation loop; travel is capped small. Static under reduced motion
// (media query) or when disabled.
import * as React from "react";
import "./Float.css";

export interface FloatProps extends React.HTMLAttributes<HTMLDivElement> {
  children?: React.ReactNode;
  /** Vertical travel in px (kept small). @default 8 */
  distance?: number;
  /** Seconds per loop. @default 4 */
  duration?: number;
  /** Lead-in delay in seconds. @default 0 */
  delay?: number;
  /** Degrees of gentle rotation at the extremes. @default 0 */
  rotate?: number;
  /** Disable (static). @default false */
  disabled?: boolean;
}

/**
 * Free Joy — Float (effect)
 * Gentle levitation loop for illustrations, empty-state art, and hero props.
 */
export function Float({
  children,
  distance = 8,
  duration = 4,
  delay = 0,
  rotate = 0,
  disabled = false,
  className = "",
  style,
  ...rest
}: FloatProps) {
  const vars = {
    "--fj-distance": `${distance}px`,
    "--fj-duration": `${duration}s`,
    "--fj-delay": `${delay}s`,
    "--fj-rotate": `${rotate}deg`,
  } as React.CSSProperties;

  const classes = ["fj-float", disabled && "fj-float--off", className].filter(Boolean).join(" ");

  return (
    <div className={classes} style={{ ...vars, ...style }} {...rest}>
      {children}
    </div>
  );
}
