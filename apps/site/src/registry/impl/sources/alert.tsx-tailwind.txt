// Free Joy — Alert (TypeScript + Tailwind CSS)
// Utilities reference the FJ tokens (arbitrary values), so every style stays token-driven.
// Tone colors are set as CSS variables on the wrapper so the static utility
// classes stay scannable by Tailwind while bg/border/text follow the tone.
"use client";

import * as React from "react";

export interface AlertProps extends React.HTMLAttributes<HTMLDivElement> {
  /** @default "info" */
  tone?: "info" | "success" | "warn" | "danger" | "neutral";
  title?: string;
  icon?: React.ReactNode;
  /** Show a dismiss button that calls this. */
  onClose?: () => void;
}

const TONES: Record<NonNullable<AlertProps["tone"]>, { bg: string; fg: string; accent: string }> = {
  info: { bg: "var(--info-100)", fg: "var(--info-700)", accent: "var(--info-500)" },
  success: { bg: "var(--success-100)", fg: "var(--success-700)", accent: "var(--success-500)" },
  warn: { bg: "var(--warn-100)", fg: "var(--warn-700)", accent: "var(--warn-500)" },
  danger: { bg: "var(--danger-100)", fg: "var(--danger-700)", accent: "var(--danger-500)" },
  neutral: { bg: "var(--paper-2)", fg: "var(--ink-2)", accent: "var(--ink-3)" },
};

/**
 * Free Joy — Alert
 * Inline message banner. tones: info | success | warn | danger | neutral.
 */
export function Alert({
  tone = "info",
  title,
  icon,
  onClose,
  className = "",
  style,
  children,
  ...rest
}: AlertProps) {
  const t = TONES[tone] ?? TONES.info;
  const toneVars = {
    "--fj-a-bg": t.bg,
    "--fj-a-fg": t.fg,
    "--fj-a-accent": t.accent,
  } as React.CSSProperties;

  return (
    <div
      role="status"
      style={{ ...toneVars, ...style }}
      className={
        "flex items-start gap-3 rounded-[var(--radius-md)] px-[15px] py-[13px] " +
        "border border-[color-mix(in_srgb,var(--fj-a-accent)_20%,transparent)] " +
        `bg-[var(--fj-a-bg)] ${className}`
      }
      {...rest}
    >
      {icon && <span className="mt-px inline-flex flex-none text-[var(--fj-a-accent)]">{icon}</span>}
      <div className="min-w-0 flex-1">
        {title && (
          <div className="mb-0.5 text-[length:var(--text-sm)] font-semibold text-[var(--fj-a-fg)]">{title}</div>
        )}
        {children && (
          <div className="text-[length:var(--text-sm)] leading-[var(--leading-normal)] text-[var(--fj-a-fg)] opacity-90">
            {children}
          </div>
        )}
      </div>
      {onClose && (
        <button
          onClick={onClose}
          aria-label="Dismiss"
          className="inline-flex flex-none cursor-pointer border-none bg-transparent p-0.5 text-[var(--fj-a-fg)] opacity-60 hover:opacity-100"
        >
          <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.2" strokeLinecap="round">
            <line x1="18" y1="6" x2="6" y2="18" />
            <line x1="6" y1="6" x2="18" y2="18" />
          </svg>
        </button>
      )}
    </div>
  );
}
