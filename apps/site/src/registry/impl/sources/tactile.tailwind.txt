// Free Joy — Tactile (effect) with Tailwind CSS.
// Pure :active press — no JS needed for the simple case. Wrap any child.

/**
 * Free Joy — Tactile (effect)
 * A satisfying press-down (scale + settle) for any child.
 */
export function Tactile({ children, scale = 0.96, duration = 120, className = "" }) {
  return (
    <span
      style={{ "--fj-scale": scale, transitionDuration: `${duration}ms` }}
      className={
        "inline-block [-webkit-tap-highlight-color:transparent] " +
        "transition-transform ease-[var(--ease-out)] " +
        "active:scale-[var(--fj-scale)] motion-reduce:transition-none motion-reduce:active:scale-100 " +
        className
      }
    >
      {children}
    </span>
  );
}
