/* Free Joy — AnimatedUnderline.css
   Pairs with AnimatedUnderline.tsx / AnimatedUnderline.jsx. Requires the FJ
   tokens (tokens/*.css).
   The underline is a background layer that grows from 0 to 100% width when
   the component adds .is-drawn (hover / inview / mount). The component feeds
   --fj-color, --fj-thickness, --fj-offset, and --fj-duration. */

.fj-underline {
  position: relative;
  display: inline-block;
  padding-bottom: calc(var(--fj-offset, 2px) + var(--fj-thickness, 2px));

  background-image: linear-gradient(var(--fj-color, var(--accent)), var(--fj-color, var(--accent)));
  background-repeat: no-repeat;
  background-position: 0 calc(100% + var(--fj-offset, 2px));
  background-size: 0% var(--fj-thickness, 2px);
  transition: background-size var(--fj-duration, 320ms) var(--ease-out);
}

.fj-underline.is-drawn {
  background-size: 100% var(--fj-thickness, 2px);
}

@media (prefers-reduced-motion: reduce) {
  /* Show a static underline so the affordance is never lost. */
  .fj-underline {
    transition: none;
    background-size: 100% var(--fj-thickness, 2px);
  }
}
