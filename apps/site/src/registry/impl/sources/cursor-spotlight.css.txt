<!-- Free Joy — CursorSpotlight (effect), no-framework version.
     Requires the FJ tokens (tokens/*.css).
     The light position follows the cursor via --fj-mx / --fj-my, set from a
     mousemove handler (~5 lines of JS, at the bottom). The radial paint, the
     clipping, and the layering are all CSS. -->

<div class="fj-cursorlight">
  <div class="fj-cursorlight-glow" aria-hidden="true"></div>
  <div class="fj-cursorlight-content">
    <!-- your grid of cards -->
    <div class="fj-card">Sketches</div>
    <div class="fj-card">Notes</div>
    <div class="fj-card">Circles</div>
    <div class="fj-card">Sets</div>
  </div>
</div>

<style>
.fj-cursorlight {
  --fj-size: 400px;
  position: relative;
  padding: var(--space-4);
  border: 1px solid var(--border);
  border-radius: var(--radius-lg);
}

.fj-cursorlight-glow {
  position: absolute;
  inset: 0;
  pointer-events: none;
  border-radius: inherit;
  background: radial-gradient(
    var(--fj-size) circle at var(--fj-mx, -9999px) var(--fj-my, -9999px),
    var(--accent-soft),
    transparent 65%
  );
  transition: background 60ms linear;
}

.fj-cursorlight-content {
  position: relative;
  display: grid;
  grid-template-columns: repeat(auto-fit, minmax(120px, 1fr));
  gap: 12px;
}

@media (prefers-reduced-motion: reduce) {
  .fj-cursorlight-glow { display: none; }
}
</style>

<script>
  document.querySelectorAll(".fj-cursorlight").forEach((box) => {
    box.addEventListener("mousemove", (e) => {
      const r = box.getBoundingClientRect();
      box.style.setProperty("--fj-mx", `${e.clientX - r.left}px`);
      box.style.setProperty("--fj-my", `${e.clientY - r.top}px`);
    });
    box.addEventListener("mouseleave", () => {
      box.style.setProperty("--fj-mx", "-9999px");
      box.style.setProperty("--fj-my", "-9999px");
    });
  });
</script>
