/* Free Joy — CursorSpotlight.css
   Pairs with CursorSpotlight.tsx / CursorSpotlight.jsx. Requires the FJ tokens
   (tokens/*.css).
   The light position follows the cursor via --fj-mx / --fj-my (set from the
   component's mousemove handler). --fj-size is the light diameter; --fj-light
   the color; --fj-intensity the center opacity. Reduced motion hides it. */

.fj-cursorlight {
  position: relative;
}

.fj-cursorlight-glow {
  position: absolute;
  inset: 0;
  pointer-events: none;
  border-radius: inherit;
  opacity: var(--fj-intensity, 1);
  background: radial-gradient(
    var(--fj-size, 400px) circle at var(--fj-mx, -9999px) var(--fj-my, -9999px),
    var(--fj-light, var(--accent-soft)),
    transparent 65%
  );
  transition: background 60ms linear;
}

.fj-cursorlight-content { position: relative; }

@media (prefers-reduced-motion: reduce) {
  .fj-cursorlight-glow { display: none; }
}
