// Free Joy — ThemeTransition (effect) with Tailwind CSS.
// The crossfade is the browser View Transitions API driven from JS; the
// ::view-transition pseudo-elements are tuned in global CSS (Tailwind can't
// target them). Add the CSS block below to your stylesheet.

/**
 * Free Joy — runThemeTransition
 * Runs a DOM update (flip data-theme) inside a View-Transitions crossfade when
 * supported and motion is allowed; otherwise applies it instantly.
 */
export function runThemeTransition(update) {
  const start = document.startViewTransition?.bind(document);
  const reduced = matchMedia("(prefers-reduced-motion: reduce)").matches;
  if (reduced || typeof start !== "function") {
    update();
    return;
  }
  start(update);
}

// Usage in a toggle button:
export function ThemeToggle() {
  const flip = () =>
    runThemeTransition(() => {
      const el = document.documentElement;
      el.dataset.theme = el.dataset.theme === "dark" ? "light" : "dark";
    });
  return (
    <button
      onClick={flip}
      className="rounded-[var(--radius-pill)] bg-[var(--surface)] px-4 py-2 text-[var(--text)] border border-[var(--border)]"
    >
      Toggle theme
    </button>
  );
}
