// Free Joy — Tabs (TypeScript + CSS)
// Pairs with Tabs.css (imported below). Requires the FJ tokens (tokens/*.css).
// The active pill is the button with [aria-selected="true"].
"use client";

import * as React from "react";
import "./Tabs.css";

export interface TabItem {
  id: string;
  label: string;
}

export interface TabsProps {
  items: TabItem[];
  /** Active tab id. Defaults to the first item. */
  value?: string;
  onChange?: (id: string) => void;
  className?: string;
}

/**
 * Free Joy — Tabs
 * items: [{ id, label }]; controlled via value + onChange (uncontrolled falls
 * back to internal state starting on the first item).
 */
export function Tabs({ items = [], value, onChange, className = "", ...rest }: TabsProps) {
  const [internal, setInternal] = React.useState(items[0]?.id);
  const active = value ?? internal;
  const select = (id: string) => {
    if (value === undefined) setInternal(id);
    onChange && onChange(id);
  };
  return (
    <div className={`fj-tabs ${className}`.trim()} role="tablist" {...rest}>
      {items.map((it) => (
        <button
          key={it.id}
          className="fj-tab"
          role="tab"
          aria-selected={it.id === active}
          onClick={() => select(it.id)}
        >
          {it.label}
        </button>
      ))}
    </div>
  );
}
