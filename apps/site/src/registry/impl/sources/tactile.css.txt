/* Free Joy — Tactile.css
   Pairs with Tactile.tsx / Tactile.jsx. Requires the FJ tokens (tokens/*.css).
   Pure CSS: :active scales the element down. Works with mouse and touch.
   The component feeds --fj-scale and --fj-duration; --off disables the press. */

.fj-tactile {
  display: inline-block;
  transition: transform var(--fj-duration, 120ms) var(--ease-out);
  -webkit-tap-highlight-color: transparent;
}

.fj-tactile:active {
  transform: scale(var(--fj-scale, 0.96));
}

.fj-tactile--off { transition: none; }
.fj-tactile--off:active { transform: none; }

@media (prefers-reduced-motion: reduce) {
  .fj-tactile { transition: none; }
  .fj-tactile:active { transform: none; }
}
