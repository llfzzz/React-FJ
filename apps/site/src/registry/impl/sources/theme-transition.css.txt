/* Free Joy — ThemeTransition.css
   Pairs with ThemeTransition.tsx / ThemeTransition.jsx. Requires the FJ tokens
   (tokens/*.css).
   Production: runThemeTransition() flips data-theme inside the browser's View
   Transitions API — the pseudo-elements below tune that crossfade and honour
   reduced motion. Unsupported browsers just swap instantly (graceful).
   The .fj-theme-flip rules style the component's scoped demo surface. */

/* Tune the default crossfade the View Transitions API applies to <html>. */
::view-transition-old(root),
::view-transition-new(root) {
  animation-duration: 320ms;
  animation-timing-function: var(--ease-in-out);
}

/* Respect reduced motion — skip the crossfade, keep the instant swap. */
@media (prefers-reduced-motion: reduce) {
  ::view-transition-old(root),
  ::view-transition-new(root) {
    animation: none;
  }
}

/* Scoped demo surface: fades out, swaps data-theme, fades back in. */
.fj-theme-flip {
  cursor: pointer;
  border-radius: var(--radius-lg);
  background: var(--bg);
  color: var(--text);
  opacity: 1;
  transition: opacity var(--fj-fade, 160ms) var(--ease-in-out);
}
.fj-theme-flip.is-fading { opacity: 0; }
.fj-theme-flip--instant { transition: none; }

@media (prefers-reduced-motion: reduce) {
  .fj-theme-flip { transition: none; }
}
