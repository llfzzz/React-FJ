/* Free Joy — Reveal.css
   Pairs with Reveal.tsx / Reveal.jsx. Requires the FJ tokens (tokens/*.css).
   Fade + rise on entry: the component adds .is-in on scroll into view.
   Cascade children carry --fj-i; --fj-delay / --fj-y / --fj-duration /
   --fj-stagger arrive from props. */

.fj-reveal-block,
.fj-reveal-item {
  transform: translateY(var(--fj-y, 14px));
  opacity: 0;
  transition:
    transform var(--fj-duration, 640ms) var(--ease-emphasized),
    opacity calc(var(--fj-duration, 640ms) * 0.7) var(--ease-out);
  transition-delay: var(--fj-delay, 0ms);
}

/* Cascade delay per child. */
.fj-reveal-item {
  transition-delay: calc(var(--fj-delay, 0ms) + var(--fj-i, 0) * var(--fj-stagger, 60ms));
}

/* Revealed state (toggled on scroll into view). */
.fj-reveal-block.is-in,
.fj-reveal-cascade.is-in .fj-reveal-item {
  transform: translateY(0);
  opacity: 1;
}

@media (prefers-reduced-motion: reduce) {
  .fj-reveal-block,
  .fj-reveal-item { transform: none; opacity: 1; transition: none; }
}
