/* Free Joy — GradientText.css
   Pairs with GradientText.tsx / GradientText.jsx. Requires the FJ tokens
   (tokens/*.css).
   The gradient is clipped to the text and slid on a loop. The component feeds
   --fj-gradient-stops, --fj-angle, and --fj-duration; .fj-gradient-text--still
   (or reduced motion) holds it still — the color still reads. */

.fj-gradient-text {
  background-image: linear-gradient(
    var(--fj-angle, 100deg),
    var(--fj-gradient-stops, var(--joy-500), var(--bloom-500), var(--sun-500), var(--joy-500))
  );
  background-size: 200% auto;
  background-clip: text;
  -webkit-background-clip: text;
  color: transparent;
  -webkit-text-fill-color: transparent;
  animation: fj-gradient-move var(--fj-duration, 6s) linear infinite;
}

.fj-gradient-text--still { animation: none; }

@keyframes fj-gradient-move {
  to { background-position: 200% center; }
}

/* Reduced motion: hold the gradient still. */
@media (prefers-reduced-motion: reduce) {
  .fj-gradient-text { animation: none; }
}
