// Free Joy — AnimatedBorder (effect) with Tailwind CSS.
// The animated @property angle + keyframes can't be expressed as utilities, so
// add the CSS block below to your global stylesheet once. The wrapper markup
// then uses utilities + the fj-angle-spin animation.

/**
 * Free Joy — AnimatedBorder (effect)
 * Hairline conic-gradient border that slowly sweeps around the panel.
 */
export function AnimatedBorder({ speed = 8, width = 1, radius = "var(--radius-lg)", children, className = "" }) {
  return (
    <div
      style={{
        padding: width,
        borderRadius: radius,
        background:
          "conic-gradient(from var(--fj-angle,0deg), var(--border) 0%, var(--joy-400), var(--bloom-300), var(--sun-300), var(--border) 100%)",
        animationDuration: `${speed}s`,
      }}
      className={`relative animate-[fj-angle-spin_linear_infinite] motion-reduce:animate-none ${className}`}
    >
      <div
        style={{ borderRadius: `calc(${radius} - ${width}px)` }}
        className="relative bg-[var(--surface)] p-[var(--space-5)]"
      >
        {children}
      </div>
    </div>
  );
}

/* Add once to your global CSS:
@property --fj-angle { syntax: "<angle>"; initial-value: 0deg; inherits: false; }
@keyframes fj-angle-spin { to { --fj-angle: 360deg; } }
*/
