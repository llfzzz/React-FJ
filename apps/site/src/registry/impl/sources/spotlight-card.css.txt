/* Free Joy — SpotlightCard.css
   Pairs with SpotlightCard.tsx / SpotlightCard.jsx. Requires the FJ tokens
   (tokens/*.css).
   The glow position follows the cursor via --fj-mx / --fj-my (set from the
   component's mousemove handler); --fj-size is the spotlight diameter.
   .fj-spotlight--static disables the hover lift + glow. */

.fj-spotlight {
  --fj-size: 320px;
  position: relative;
  overflow: hidden;
  background: var(--surface);
  border: 1px solid var(--border);
  border-radius: var(--radius-lg);
  padding: var(--space-5);
  box-shadow: var(--shadow-xs);
  transition:
    box-shadow var(--dur-base) var(--ease-out),
    transform var(--dur-base) var(--ease-out);
}
.fj-spotlight:hover {
  box-shadow: var(--shadow-md);
  transform: translateY(-2px);
}
.fj-spotlight--static:hover {
  box-shadow: var(--shadow-xs);
  transform: none;
}

.fj-spotlight-glow {
  position: absolute;
  inset: 0;
  pointer-events: none;
  opacity: 0;
  transition: opacity var(--dur-base) var(--ease-out);
  background: radial-gradient(
    var(--fj-size) circle at var(--fj-mx, 50%) var(--fj-my, 40%),
    var(--accent-soft),
    transparent 70%
  );
}
.fj-spotlight:hover .fj-spotlight-glow { opacity: 1; }
.fj-spotlight--static:hover .fj-spotlight-glow { opacity: 0; }

.fj-spotlight-content { position: relative; }

@media (prefers-reduced-motion: reduce) {
  .fj-spotlight { transition-duration: 0.01ms; }
  .fj-spotlight:hover { transform: none; }
}
