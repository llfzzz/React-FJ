/* Free Joy — GridPattern.css
   Pairs with GridPattern.tsx / GridPattern.jsx. Requires the FJ tokens
   (tokens/*.css).
   A static dot (or line, via --line) grid with an optional radial fade mask
   (--fade). The component feeds --fj-size, --fj-color, and --fj-opacity.
   No animation. */

.fj-grid {
  position: relative;
}

.fj-grid::before {
  content: "";
  position: absolute;
  inset: 0;
  pointer-events: none;
  opacity: var(--fj-opacity, 0.5);

  /* Dot grid (default) */
  background-image: radial-gradient(var(--fj-color, var(--border)) 1px, transparent 1px);
  background-size: var(--fj-size, 24px) var(--fj-size, 24px);
}

/* Line grid */
.fj-grid--line::before {
  background-image:
    linear-gradient(to right, var(--fj-color, var(--border)) 1px, transparent 1px),
    linear-gradient(to bottom, var(--fj-color, var(--border)) 1px, transparent 1px);
}

/* Radial fade so the grid dissolves toward the edges. */
.fj-grid--fade::before {
  -webkit-mask-image: radial-gradient(ellipse at center, #000 40%, transparent 75%);
  mask-image: radial-gradient(ellipse at center, #000 40%, transparent 75%);
}

.fj-grid-content { position: relative; }
