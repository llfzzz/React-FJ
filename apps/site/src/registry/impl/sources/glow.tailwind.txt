// Free Joy — Glow (effect) with Tailwind CSS.
// Utilities reference the FJ tokens (arbitrary values) so both themes work.
// Static single blur layer — no animation, cheap.

/**
 * Free Joy — Glow (effect)
 * Static soft gradient glow behind a child element.
 */
export function Glow({ colors = ["var(--joy-300)", "var(--bloom-300)"], intensity = 0.35, blur = 48, spread = 8, className = "", children }) {
  return (
    <div className={`relative inline-block ${className}`}>
      <div
        aria-hidden="true"
        style={{
          inset: -spread,
          background: `linear-gradient(120deg, ${colors.join(", ")})`,
          filter: `blur(${blur}px)`,
          opacity: intensity,
        }}
        className="pointer-events-none absolute rounded-[var(--radius-xl)]"
      />
      <div className="relative">{children}</div>
    </div>
  );
}
