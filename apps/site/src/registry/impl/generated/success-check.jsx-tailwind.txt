// Free Joy — SuccessCheck (effect) (JavaScript + Tailwind CSS)
// The stroke-draw is a transition on stroke-dashoffset; the ring pulse needs
// one keyframes rule — add the CSS block below to your global stylesheet.
// Reduced motion shows the finished check instantly.
"use client";

import * as React from "react";

/**
 * Free Joy — SuccessCheck (effect)
 * An SVG checkmark that draws itself in, with an optional soft ring — for
 * confirmations and completed steps.
 */
export function SuccessCheck({
  size = 48,
  color = "var(--success-500)",
  duration = 500,
  ring = true,
  trigger = "mount",
  active,
  label = "Success",
  className = "",
  style,
  ...rest
}) {
  const ref = React.useRef(null);
  const [drawn, setDrawn] = React.useState(false);

  React.useEffect(() => {
    if (trigger === "mount") {
      const id = requestAnimationFrame(() => setDrawn(true));
      return () => cancelAnimationFrame(id);
    }
    if (trigger === "inview") {
      const el = ref.current;
      if (!el) return;
      const io = new IntersectionObserver(
        ([e]) => {
          if (e.isIntersecting) {
            setDrawn(true);
            io.disconnect();
          }
        },
        { threshold: 0.5 },
      );
      io.observe(el);
      return () => io.disconnect();
    }
  }, [trigger]);

  const on = trigger === "manual" ? Boolean(active) : drawn;

  return (
    <span
      ref={ref}
      role="img"
      aria-label={label}
      style={{ width: size, height: size, color, ...style }}
      className={`relative inline-flex ${className}`}
      {...rest}
    >
      {ring && on && (
        <span
          aria-hidden="true"
          className="absolute inset-0 rounded-full border-2 border-current animate-[fj-check-ring_700ms_var(--ease-out)_forwards] motion-reduce:hidden"
        />
      )}
      <svg viewBox="0 0 52 52" aria-hidden="true" className="size-full">
        <circle cx="26" cy="26" r="24" fill="none" stroke="currentColor" strokeOpacity="0.2" strokeWidth="2" />
        <path
          d="M14 27 L23 35 L38 18"
          fill="none"
          stroke="currentColor"
          strokeWidth="3.5"
          strokeLinecap="round"
          strokeLinejoin="round"
          className="motion-reduce:!transition-none motion-reduce:![stroke-dashoffset:0]"
          style={{
            strokeDasharray: 28,
            strokeDashoffset: on ? 0 : 28,
            transition: `stroke-dashoffset ${duration}ms var(--ease-out)`,
          }}
        />
      </svg>
    </span>
  );
}

/* Add once to your global CSS:
@keyframes fj-check-ring { from { transform: scale(0.7); opacity: 0.6; } to { transform: scale(1.15); opacity: 0; } }
