// Free Joy — Shimmer (effect) (JavaScript + Tailwind CSS)
// The sweep keyframes can't be a utility — add the CSS block below to your
// global stylesheet once. The overlay then uses utilities + fj-shimmer-sweep;
// hover / mount triggers ride the animation play-state and iteration count.

/**
 * Free Joy — Shimmer (effect)
 * A soft light sheen sweeps across a surface — badges, skeleton chips, freshly
 * loaded cards. Wraps its children; the sheen is a pointer-transparent overlay.
 */
export function Shimmer({
  children,
  trigger = "loop",
  duration = 1400,
  angle = 20,
  intensity = 0.35,
  disabled = false,
  className = "",
  ...rest
}) {
  return (
    <span className={`group relative inline-block overflow-hidden ${className}`} {...rest}>
      {children}
      {!disabled && (
        <span
          aria-hidden="true"
          style={{
            background: `linear-gradient(${90 + angle}deg, transparent 20%, color-mix(in srgb, var(--white) ${Math.round(
              intensity * 100,
            )}%, transparent) 50%, transparent 80%)`,
            animationDuration: `${duration}ms`,
            animationIterationCount: trigger === "mount" ? 1 : undefined,
          }}
          className={
            "pointer-events-none absolute inset-0 -translate-x-full " +
            "animate-[fj-shimmer-sweep_var(--ease-in-out)_infinite] " +
            "motion-reduce:animate-none motion-reduce:opacity-0 " +
            (trigger === "hover"
              ? "[animation-play-state:paused] group-hover:[animation-play-state:running]"
              : "")
          }
        />
      )}
    </span>
  );
}

/* Add once to your global CSS:
@keyframes fj-shimmer-sweep { from { transform: translateX(-100%); } to { transform: translateX(100%); } }
*/
