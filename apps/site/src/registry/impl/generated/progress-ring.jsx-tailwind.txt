// Free Joy — ProgressRing (effect) (JavaScript + Tailwind CSS)
// The sweep is a plain CSS transition on stroke-dashoffset — no keyframes
// needed. The trigger (mount / inview / manual) flips the value from 0 to its
// target so the transition sweeps in. Reduced motion jumps to the value.
"use client";

import * as React from "react";

/**
 * Free Joy — ProgressRing (effect)
 * A circular progress ring that sweeps to its value — goals, scores, upload
 * completion. Later value changes animate through the same transition.
 */
export function ProgressRing({
  value,
  size = 48,
  strokeWidth = 4,
  color = "var(--accent)",
  duration = 600,
  showValue = false,
  trigger = "mount",
  active,
  label = "Progress",
  className = "",
  style,
  ...rest
}) {
  const ref = React.useRef(null);
  const [on, setOn] = React.useState(trigger === "mount");

  React.useEffect(() => {
    if (trigger !== "inview") return;
    const el = ref.current;
    if (!el) return;
    const io = new IntersectionObserver(
      ([e]) => {
        if (e.isIntersecting) {
          setOn(true);
          io.disconnect();
        }
      },
      { threshold: 0.5 },
    );
    io.observe(el);
    return () => io.disconnect();
  }, [trigger]);

  const clamped = Math.max(0, Math.min(100, value));
  const r = (size - strokeWidth) / 2;
  const c = 2 * Math.PI * r;
  const shown = (trigger === "manual" ? Boolean(active) : on) ? clamped : 0;

  return (
    <span
      ref={ref}
      role="progressbar"
      aria-valuemin={0}
      aria-valuemax={100}
      aria-valuenow={Math.round(clamped)}
      aria-label={label}
      style={{ width: size, height: size, color, ...style }}
      className={`relative inline-flex ${className}`}
      {...rest}
    >
      <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`} aria-hidden="true">
        <circle cx={size / 2} cy={size / 2} r={r} fill="none" stroke="currentColor" strokeOpacity={0.15} strokeWidth={strokeWidth} />
        <circle
          cx={size / 2}
          cy={size / 2}
          r={r}
          fill="none"
          stroke="currentColor"
          strokeWidth={strokeWidth}
          strokeLinecap="round"
          style={{ strokeDasharray: c, strokeDashoffset: c * (1 - shown / 100) }}
          className="origin-center -rotate-90 transition-[stroke-dashoffset] duration-[600ms] ease-[var(--ease-out)] motion-reduce:transition-none"
        />
      </svg>
      {showValue && (
        <span aria-hidden="true" className="absolute inset-0 grid place-items-center text-xs font-semibold tabular-nums text-[var(--ink-1)]">
          {Math.round(clamped)}%
        </span>
      )}
    </span>
  );
}
