// Free Joy — Drawer (JavaScript + CSS)
// Pairs with Drawer.css (imported below). Requires the FJ tokens (tokens/*.css).
// Slide-in panel anchored to any edge, with a scrim. Traps focus, restores it
// on close, closes on Escape. The panel size rides --fj-drawer-size.
"use client";

import * as React from "react";
import "./Drawer.css";

/**
 * Free Joy — Drawer
 * Slide-in panel anchored to any edge, with a scrim. Controlled via `open`.
 */
export function Drawer({
  open,
  onClose,
  side = "right",
  size = 380,
  title,
  children,
  footer,
  className = "",
}) {
  const panelRef = React.useRef(null);

  React.useEffect(() => {
    if (!open) return;
    const prev = document.activeElement;
    const sel = 'a[href], button:not([disabled]), textarea, input, select, [tabindex]:not([tabindex="-1"])';
    const panel = panelRef.current;
    const first = panel?.querySelector(sel);
    (first || panel)?.focus();
    const onKey = (e) => {
      if (e.key === "Escape") onClose?.();
      else if (e.key === "Tab" && panel) {
        const items = Array.from(panel.querySelectorAll(sel)).filter((el) => el.offsetParent !== null);
        if (!items.length) {
          e.preventDefault();
          return;
        }
        const firstEl = items[0];
        const lastEl = items[items.length - 1];
        if (e.shiftKey && document.activeElement === firstEl) {
          e.preventDefault();
          lastEl.focus();
        } else if (!e.shiftKey && document.activeElement === lastEl) {
          e.preventDefault();
          firstEl.focus();
        }
      }
    };
    document.addEventListener("keydown", onKey);
    return () => {
      document.removeEventListener("keydown", onKey);
      prev?.focus?.();
    };
  }, [open, onClose]);

  const classes = ["fj-drawer", `fj-drawer--${side}`, open && "is-open", className]
    .filter(Boolean)
    .join(" ");

  return (
    <div className={classes} aria-hidden={!open}>
      <div className="fj-drawer-scrim" onClick={onClose} />
      <aside
        className="fj-drawer-panel"
        role="dialog"
        aria-modal="true"
        aria-label={typeof title === "string" ? title : undefined}
        ref={panelRef}
        tabIndex={-1}
        style={{ "--fj-drawer-size": `${size}px` }}
      >
        {(title || onClose) && (
          <div className="fj-drawer-head">
            <span className="fj-drawer-title">{title}</span>
            {onClose && (
              <button className="fj-drawer-close" onClick={onClose} aria-label="Close">
                ×
              </button>
            )}
          </div>
        )}
        <div className="fj-drawer-body">{children}</div>
        {footer && <div className="fj-drawer-foot">{footer}</div>}
      </aside>
    </div>
  );
}
