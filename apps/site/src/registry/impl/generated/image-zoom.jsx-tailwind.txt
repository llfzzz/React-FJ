// Free Joy — ImageZoom (effect) (JavaScript + Tailwind CSS)
// Hover mode is pure utilities (group-hover scale inside an overflow-hidden
// frame). The drift (ken-burns) keyframes can't be utilities — add the CSS
// block below to your global stylesheet once. Keep scale ≤ 1.15.

const MAX_SCALE = 1.15; // keep the zoom subtle — motion-policy cap

/**
 * Free Joy — ImageZoom (effect)
 * A gentle zoom inside a cropped frame — hover zoom for gallery cards, or a
 * slow ken-burns drift for hero imagery.
 */
export function ImageZoom({
  children,
  mode = "hover",
  scale = 1.06,
  duration = 600,
  speed = 1,
  disabled = false,
  className = "",
  style,
  ...rest
}) {
  const effScale = Math.min(scale, MAX_SCALE);
  return (
    <div className={`group overflow-hidden rounded-[var(--radius-lg)] ${className}`} style={style} {...rest}>
      <div
        style={
          mode === "drift"
            ? ({ "--fj-scale": effScale, animationDuration: `${16 / speed}s` })
            : { transitionDuration: `${duration}ms` }
        }
        className={
          disabled
            ? ""
            : mode === "drift"
              ? "animate-[fj-kenburns_var(--ease-in-out)_infinite_alternate] motion-reduce:animate-none"
              : "transition-transform ease-[var(--ease-out)] " +
                `group-hover:scale-[${effScale}] ` +
                "motion-reduce:transition-none motion-reduce:group-hover:scale-100"
        }
      >
        {children}
      </div>
    </div>
  );
}
