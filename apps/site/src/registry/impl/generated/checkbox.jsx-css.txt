// Free Joy — Checkbox (JavaScript + CSS)
// Pairs with Checkbox.css (imported below). Requires the FJ tokens (tokens/*.css).
// The native input is visually hidden; the sibling box is the styled surface,
// and :checked drives its fill + tick (the label wraps both for a wide hit target).
"use client";

import * as React from "react";
import "./Checkbox.css";

/** Free Joy — Checkbox */
export function Checkbox({
  checked = false,
  onChange,
  disabled = false,
  label,
  id,
  className = "",
  ...rest
}) {
  const autoId = React.useId();
  const fid = id || autoId;
  return (
    <label className={`fj-check ${className}`.trim()} htmlFor={fid}>
      <input
        className="fj-check-input"
        id={fid}
        type="checkbox"
        checked={checked}
        disabled={disabled}
        onChange={(e) => onChange && onChange(e.target.checked, e)}
        {...rest}
      />
      <span className="fj-check-box" aria-hidden="true">
        <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="3.5" strokeLinecap="round" strokeLinejoin="round">
          <polyline points="20 6 9 17 4 12" />
        </svg>
      </span>
      {label && <span className="fj-check-label">{label}</span>}
    </label>
  );
}
