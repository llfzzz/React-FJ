// Free Joy — Waves (effect) (JavaScript + CSS)
// Pairs with Waves.css (imported below). Requires the FJ tokens (tokens/*.css).
// Slow wave lines drifting across a band at the container's bottom edge. The
// geometry is drawn once (a two-period path); the stylesheet's keyframe slides
// exactly one period, so the loop is seamless. Only transform animates — never
// the path. Static under reduced motion (media query).
"use client";

import * as React from "react";
import "./Waves.css";

// One period of the wave in viewBox units. The path spans TWO periods and the
// keyframe slides exactly ONE, so the loop is seamless. These constants must
// stay in sync with Waves.css (the keyframe translates by -PERIOD).
const PERIOD = 1440;
const VIEW_H = 120;
const MAX_LAYERS = 4;
const MAX_AMPLITUDE = 24;
const OPACITIES = [0.38, 0.3, 0.22, 0.14];

/** A smooth two-period sine-ish stroke path (Q + reflected T segments). */
function wavePath(amplitude) {
  const y = VIEW_H / 2;
  const half = PERIOD / 4; // half a bump
  let d = `M 0 ${y} Q ${half / 2} ${y - amplitude} ${half} ${y}`;
  for (let x = half * 2; x <= PERIOD * 2; x += half) {
    d += ` T ${x} ${y}`;
  }
  return d;
}

;

/**
 * Free Joy — Waves (effect)
 * Slow wave lines drifting across a band at the container's bottom edge — a
 * quiet texture for heroes and footers.
 */
export function Waves({
  children,
  layers = 3,
  color = "var(--joy-300)",
  amplitude = 14,
  height = 120,
  speed = 1,
  performance = "full",
  disabled = false,
  className = "",
  style,
  ...rest
}) {
  const n = Math.min(performance === "lite" ? 2 : MAX_LAYERS, Math.max(1, layers));
  const d = React.useMemo(() => wavePath(Math.min(amplitude, MAX_AMPLITUDE)), [amplitude]);

  return (
    <div
      className={`fj-waves ${disabled ? "fj-waves--static " : ""}${className}`.trim()}
      style={style}
      {...rest}
    >
      <svg
        className="fj-waves-band"
        aria-hidden="true"
        viewBox={`0 0 ${PERIOD} ${VIEW_H}`}
        preserveAspectRatio="none"
        style={{ height }}
      >
        {Array.from({ length: n }, (_, i) => (
          <g
            key={i}
            className="fj-wave"
            style={{
              animationDuration: `${(12 + i * 5) / speed}s`,
              animationDirection: i % 2 === 1 ? "reverse" : undefined,
            }}
          >
            <path
              d={d}
              transform={`translate(0 ${(i - (n - 1) / 2) * 14})`}
              fill="none"
              stroke={color}
              strokeWidth={2}
              strokeOpacity={OPACITIES[i]}
              vectorEffect="non-scaling-stroke"
            />
          </g>
        ))}
      </svg>
      <div className="fj-waves-content">{children}</div>
    </div>
  );
}
