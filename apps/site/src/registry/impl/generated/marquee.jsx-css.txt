// Free Joy — Marquee (effect) (JavaScript + CSS)
// Pairs with Marquee.css (imported below). Requires the FJ tokens (tokens/*.css).
// The strip is rendered twice; the track slides one strip-width (-50%) on a
// linear loop, so the seam is invisible. The clone is aria-hidden + inert.
// Hover pauses; reduced motion stops the loop AND hides the clone (media query).
"use client";

import "./Marquee.css";

/**
 * Free Joy — Marquee (effect)
 * An infinite, seamless ticker for decorative streams — logo walls, tag clouds.
 */
export function Marquee({
  children,
  duration = 24,
  direction = "left",
  pauseOnHover = true,
  gap = 24,
  fade = true,
  disabled = false,
  className = "",
  style,
  ...rest
}) {
  const group = (clone) => (
    <div
      className="fj-marquee-group"
      aria-hidden={clone || undefined}
      inert={clone || undefined}
      style={{ "--fj-gap": `${gap}px` }}
    >
      {children}
    </div>
  );

  const classes = [
    "fj-marquee",
    !fade && "fj-marquee--no-fade",
    !pauseOnHover && "fj-marquee--no-pause",
    disabled && "fj-marquee--static",
    className,
  ]
    .filter(Boolean)
    .join(" ");

  return (
    <div className={classes} style={style} {...rest}>
      <div
        className="fj-marquee-track"
        style={
          {
            "--fj-duration": `${duration}s`,
            animationDirection: direction === "right" ? "reverse" : undefined,
          }
        }
      >
        {group(false)}
        {group(true)}
      </div>
    </div>
  );
}
