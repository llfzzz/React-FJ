// Free Joy — Float (effect) (JavaScript + Tailwind CSS)
// The float keyframes depend on runtime distance/rotate, so add a parametric
// keyframes rule to your global CSS (below) and drive it with CSS variables.

/**
 * Free Joy — Float (effect)
 * Gentle levitation loop for illustrations, empty-state art, and hero props.
 * Static under reduced motion or disabled.
 */
export function Float({
  children,
  distance = 8,
  duration = 4,
  delay = 0,
  rotate = 0,
  disabled = false,
  className = "",
  style,
  ...rest
}) {
  return (
    <div
      style={{
        "--fj-distance": `${distance}px`,
        "--fj-rotate": `${rotate}deg`,
        animationDuration: `${duration}s`,
        animationDelay: `${delay}s`,
        ...style,
      }}
      className={
        "inline-block will-change-transform " +
        (disabled ? "" : "animate-[fj-float_var(--ease-in-out)_infinite] motion-reduce:animate-none ") +
        className
      }
      {...rest}
    >
      {children}
    </div>
  );
}
