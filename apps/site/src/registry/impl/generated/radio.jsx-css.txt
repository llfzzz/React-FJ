// Free Joy — Radio / RadioGroup (JavaScript + CSS)
// Pairs with Radio.css (imported below). Requires the FJ tokens (tokens/*.css).
// Native radios are visually hidden; :checked scales in the inner dot.
"use client";

import * as React from "react";
import "./Radio.css";

/**
 * Free Joy — Radio / RadioGroup
 * Single-choice control. Use RadioGroup for a set; Radio for one option.
 */
export function Radio({
  checked = false,
  onChange,
  disabled = false,
  label,
  name,
  value,
  id,
  className = "",
  ...rest
}) {
  const autoId = React.useId();
  const fid = id || autoId;
  return (
    <label className={`fj-radio ${className}`.trim()} htmlFor={fid}>
      <input
        className="fj-radio-input"
        id={fid}
        type="radio"
        name={name}
        value={value}
        checked={checked}
        disabled={disabled}
        onChange={(e) => onChange && onChange(value ?? "", e)}
        {...rest}
      />
      <span className="fj-radio-ring" aria-hidden="true">
        <span className="fj-radio-dot" />
      </span>
      {label && <span className="fj-radio-label">{label}</span>}
    </label>
  );
}

export function RadioGroup({ value, onChange, options = [], name, className = "", ...rest }) {
  const autoName = React.useId();
  const gname = name || autoName;
  return (
    <div role="radiogroup" className={`fj-radiogroup ${className}`.trim()} {...rest}>
      {options.map((o) => {
        const opt = typeof o === "string" ? { label: o, value: o } : o;
        return (
          <Radio
            key={opt.value}
            name={gname}
            value={opt.value}
            label={opt.label}
            checked={value === opt.value}
            onChange={onChange}
          />
        );
      })}
    </div>
  );
}
