// Free Joy — ScrollProgress (effect) (JavaScript + CSS)
// Pairs with ScrollProgress.css (imported below). Requires the FJ tokens
// (tokens/*.css). The fill reflects the scroll position (JS state); only
// scaleX changes — no layout, no reflow. Reduced-motion safe: it reflects
// position, it isn't a decorative animation.
"use client";

import * as React from "react";
import "./ScrollProgress.css";

/**
 * Free Joy — ScrollProgress (effect)
 * A thin reading-progress bar pinned to the top (or bottom) of the viewport
 * that fills as the page scrolls.
 */
export function ScrollProgress({
  height = 3,
  color = "var(--accent)",
  position = "top",
  target,
  zIndex,
  className = "",
  style,
  ...rest
}) {
  const [progress, setProgress] = React.useState(0);

  React.useEffect(() => {
    const el = target?.current;
    const compute = () => {
      if (el) {
        const max = el.scrollHeight - el.clientHeight;
        setProgress(max > 0 ? el.scrollTop / max : 0);
      } else {
        const max = document.documentElement.scrollHeight - window.innerHeight;
        setProgress(max > 0 ? window.scrollY / max : 0);
      }
    };
    const src = el ?? window;
    compute();
    src.addEventListener("scroll", compute, { passive: true });
    window.addEventListener("resize", compute);
    return () => {
      src.removeEventListener("scroll", compute);
      window.removeEventListener("resize", compute);
    };
  }, [target]);

  const classes = [
    "fj-scrollbar",
    position === "bottom" && "fj-scrollbar--bottom",
    target && "fj-scrollbar--sticky",
    className,
  ]
    .filter(Boolean)
    .join(" ");

  const vars = {
    "--fj-height": `${height}px`,
    "--fj-color": color,
    ...(zIndex !== undefined ? { "--fj-z": zIndex } : {}),
  };

  return (
    <div
      className={classes}
      role="progressbar"
      aria-valuemin={0}
      aria-valuemax={100}
      aria-valuenow={Math.round(progress * 100)}
      aria-label="Reading progress"
      style={{ ...vars, transform: `scaleX(${progress})`, ...style }}
      {...rest}
    />
  );
}
