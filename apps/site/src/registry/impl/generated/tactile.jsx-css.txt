// Free Joy — Tactile (effect) (JavaScript + CSS)
// Pairs with Tactile.css (imported below). Requires the FJ tokens (tokens/*.css).
// Pure CSS: :active scales the element down. Works with mouse and touch.
// Under reduced motion the press is instant with no scaling (media query).
import "./Tactile.css";

/**
 * Free Joy — Tactile (effect)
 * Adds a satisfying press-down (scale + settle) to any child — buttons, cards,
 * tappable tiles.
 */
export function Tactile({
  children,
  scale = 0.96,
  duration = 120,
  disabled = false,
  className = "",
  style,
  ...rest
}) {
  const classes = ["fj-tactile", disabled && "fj-tactile--off", className]
    .filter(Boolean)
    .join(" ");

  return (
    <span
      className={classes}
      style={{ "--fj-scale": scale, "--fj-duration": `${duration}ms`, ...style }}
      {...rest}
    >
      {children}
    </span>
  );
}
