// Free Joy — AnimatedUnderline (effect) (JavaScript + Tailwind CSS)
// Utilities reference the FJ tokens (arbitrary values), so every style stays token-driven.
// The underline is a background layer; its width is state-driven (hover /
// inview / mount), so the size rides inline style while the transition is a
// utility. Reduced motion shows a static underline immediately.
"use client";

import * as React from "react";

/**
 * Free Joy — AnimatedUnderline (effect)
 * A restrained underline that draws in from the left on hover or scroll.
 * Safe for docs links and inline emphasis.
 */
export function AnimatedUnderline({
  children,
  trigger = "hover",
  color = "var(--accent)",
  thickness = 2,
  offset = 2,
  duration = 320,
  disabled = false,
  className = "",
  style,
  ...rest
}) {
  const ref = React.useRef(null);
  const [active, setActive] = React.useState(trigger === "mount");

  React.useEffect(() => {
    if (trigger !== "inview") return;
    const el = ref.current;
    if (!el) return;
    const io = new IntersectionObserver(
      ([e]) => {
        if (e.isIntersecting) {
          setActive(true);
          io.disconnect();
        }
      },
      { threshold: 0.5 },
    );
    io.observe(el);
    return () => io.disconnect();
  }, [trigger]);

  const drawn = disabled || active;

  return (
    <span
      ref={ref}
      onMouseEnter={trigger === "hover" ? () => setActive(true) : undefined}
      onFocus={trigger === "hover" ? () => setActive(true) : undefined}
      style={{
        backgroundImage: `linear-gradient(${color}, ${color})`,
        backgroundPosition: `0 calc(100% + ${offset}px)`,
        backgroundSize: `${drawn ? 100 : 0}% ${thickness}px`,
        paddingBottom: offset + thickness,
        transitionDuration: `${duration}ms`,
        // Read by the motion-reduce fallback class below.
        "--fj-th": `${thickness}px`,
        ...style,
      }}
      className={
        "relative inline-block bg-no-repeat " +
        "transition-[background-size] ease-[var(--ease-out)] " +
        "motion-reduce:transition-none motion-reduce:bg-[length:100%_var(--fj-th)] " +
        className
      }
      {...rest}
    >
      {children}
    </span>
  );
}
